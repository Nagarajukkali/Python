#Write an interactive program that when given an input string by the user like "500 plus 200 equals" will output 700.

print ("Input string by the user like "500 plus 200 equals" will output 700.")
num1=input("Enter first number as 500")
num2=input("Enter Second number as 200")

#Adding 2 number
sum=int(num1)+int(num2);

# Display the sum
print('The sum of {0} and {1} is {2}'.format(num1, num2, sum))
